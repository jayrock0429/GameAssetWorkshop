    def _get_style_hint(self, image_path):
        """[V11.2]  AI Style Hint """
        gemini_key = os.environ.get("GEMINI_API_KEY")
        if not gemini_key or not os.path.exists(image_path):
            return "Professional Slot Art"
            
        try:
             from google import genai
             from google.genai import types
             client = genai.Client(api_key=gemini_key)
             
             # Load image for processing
             with open(image_path, 'rb') as f:
                 img_data = f.read()
                 
             prompt = "Describe the visual style of this image in 5 keywords for an AI artist (e.g., 'gold neon, 3D render, cyberpunk')."
             
             response = client.models.generate_content(
                 model='gemini-2.5-flash-image',
                 contents=[
                     prompt,
                     types.Part.from_bytes(data=img_data, mime_type="image/png")
                 ]
             )
             return response.text.strip()
        except Exception as e:
            print(f"WARNING: Style Hint Generation failed: {e}")
            return "Professional Slot Art"
